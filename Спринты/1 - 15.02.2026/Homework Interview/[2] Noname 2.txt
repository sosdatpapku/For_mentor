# Задача №4 (Python)

def rgb_sort(color_list):
    red_list, green_list, blue_list = [], [], []
    for i in color_list:
        if i.startswith('r'):
            red_list.append(i)
        elif i.startswith('g'):
            green_list.append(i)
        elif i.startswith('b'):
            blue_list.append(i)
    red_list.sort()
    green_list.sort()
    blue_list.sort()

    return red_list + green_list + blue_list
